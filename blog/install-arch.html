How to Install arch linux

  

Connect to WiFi 

iwctl
station wlan0 conncet "name of the network"
timedatectl set-ntp true
pacman -Sy archlinux-keyring 
pacman -Sy reflector
reflector -c India -a 6 --sort rate --save /etc/pacman.d/mirrorlist

  Create Partition 

cfdisk
     
 mkfs.ext4 /dev/root_partition
mkfs.fat -F 32 /dev/efi-partition 
mkswap /dev/swap_partition
swapon /dev/swap_partition
mount /dev/root_partition /mnt
mount --mkdir /dev/efi_system_partition /mnt/boot/efi

Installtion of kenel
 

 pacstrap /mnt base linux linux-firmware nano intel-ucode grub efibootmgr networkmanager sudo  

Disk  setup 

  genfstab -U /mnt
genfstab -U /mnt > /mnt/etc/fstab 
  arch-chroot /mnt 
lsblk
blkid (for check UUID of drive )
 optinal
 fallocate -l 2GB /swapfile
 chmod 600 /swapfile
 mkswap /swapfile
swapon /swapfile
 nano /etc/fsatb (put) /swapfile none swap defaults 0 0

Time setup 

  ln -sf /usr/share/zoneinfo/Region/City /etc/localtime
hwclock --systohc  

Langauage set-up 

nano /etc/locale.gen (uncomment this line)  en_US.UTF-8 UTF-8
locale-gen 
nano /etc/locale.conf (and put this) LANG=en_US.UTF-8
or --> echo "LANG=en_US.UTF-8" >> /etc/locale.conf
 nano /etc/hostname (and put this) delta
 nano /etc/hosts (and put below 3 lines)
127.0.0.1  (tab) localhost
::1 (2xtab) localhost
127.0.1.1 (tab) delta.localdomain delta

 Some impotant pkg 

pacman -Sy dialog dhcpcd wpa_supplicant ifplugd netctl 

 Add user 


useradd -G wheel,audio,video -m gamma
passwd gamma
ls /home/gamma (check)
nano /etc/sudoers
Uncomment this line  --> %wheel      ALL=(ALL:ALL) ALL
pacman -S grub efibootmgr ntfs-3g networkmanager network-manager-applet wireless_tools wpa_supplicant dialog mtools dosfstools base-devel linux-headers git bluez bluez-utils pulseaudio cups os-prober

To install grub remember some important points  

1) drive format --> mkfs.fat -F 32 /dev/efi-partition 
2)UEFI  or EFI MODE ON 
3)check mount drive fstab and blkid 
4)if grub only showing UEFI than reinstall linux kernel
Chroot Pacman -S linux 
grub-install --target=x86_64-efi --efi-directory=/efi/ --bootloader-id=GRUB
grub-mkconfig -o /boot/grub/grub.cfg
systemctl enable NetworkManager 
exit
reboot 

 Grub problem. ---> enabled efimode or UEFI

1) mkfs.fat -F32 /dev/efi-partition
2) modprobe efivarfs
3)grub-install --target=x86_64-efi --efi-directory=/efi/ --bootloader-id=GRUB --recheck --debug
In chroot --> mount /dev/efi-partition 
grub-mkconfig -o /boot/grub/grub.cfg
passwd (root user pass) 
exit
reboot
